变量关键字

  • var ：仅仅声明一个变量，不初始化内存地址
  • make/new相同点：声明变量，又初始化内存地址
  • make/new不同点
    • make：返回声明变量的本身（slice、map、array）
    • new：返回变量的内存地址（指针）（struct）

json和map

Let’s see one more example where we convert a map to a JSON where we have a struct for the value in the map. Below is the code for that

package main
import (
    "encoding/json"
    "fmt"
)
type employee struct {
    Name string
}
func main() {
    a := make(map[string]employee)
    a["1"] = employee{Name: "John"}
    j, err := json.Marshal(a)
    if err != nil {
        fmt.Printf("Error: %s", err.Error())
    } else {
        fmt.Println(string(j))
    }
}

Output

{"1":{"Name":"John"}}

集合map

Map 是一个无序的 key/value 集合；

Map 中所有的 key 都是不同的；

通过给定的 key ，可以在常数时间复杂度内查找、更新或删除相应的 value。

map每次遍历时，key的顺序都是不同的。
业务依赖key次序时，如何解决随机化问题
其实Go maps in action一文已经给出了解决方法：
If you require a stable iteration order you must maintain a separate data structure that specifies that order.
可见，需要另外维护一个数据结构来保持有序的key，然后根据有序key来遍历map。 

package main

import (
    "fmt"
    "reflect"
    "sort"
)

func main() {

    //初始化map
    //m := map[string]string{}
    m := make(map[string]string)

    //赋值
    m["a"] = "123"
    m["b"] = "456"
    m["c"] = "789"

    //获取类型
    fmt.Printf("类型是：%s\n", reflect.TypeOf(m))
    fmt.Printf("类型是：%T\n", m)

    //打印单个元素
    fmt.Println(m["b"])

    //删除元素
    delete(m, "a")
    m["a"] = "1"

    //遍历
    fmt.Println("排序前遍历：")
    for k, v := range m {
        fmt.Printf("deteil: k=%s, v=%s\n", k, v)
    }

    fmt.Println("\n排序后遍历：")
    //新生成一个数组用于存放key
    keys := make([]string, 0)
    //keys := []string{}

    for k, _ := range m {
        keys = append(keys, k)
    }
    sort.Strings(keys)
    for _, k := range keys {
        fmt.Printf("sorted: k=%s, v=%s\n", k, m[k])
    }

}

func contains(s []string, ele string) bool {
    for _, v := range s {
        if v == ele {
            return true
        }
    }
    return false
}

func uniqSlice(s []string) []string {
    var res []string
    for _, v := range s {
        if contains(res, v) == false {
            res = append(res, v)
        }
    }
    return res
}

切片中的集合排序

type User struct {
    Name string
    Age  int
}

func main() {
    // Sorting a slice of structs by a field
    users := []User{
        {
            Name: "Rajeev",
            Age:  28,
        },
        {
            Name: "Monica",
            Age:  31,
        },
        {
            Name: "John",
            Age:  56,
        },
        {
            Name: "Amanda",
            Age:  16,
        },
        {
            Name: "Steven",
            Age:  28,
        },
    }

    // Sort users by their age
    sort.Slice(users, func(i, j int) bool {
        return users[i].Age < users[j].Age
    })
    fmt.Println("Sorted users by age: ", users)

    // Stable sort
    sort.SliceStable(users, func(i, j int) bool {
        return users[i].Age < users[j].Age
    })
    fmt.Println("Sorted users by age: ", users)
}

package main

func main() {
    //初始化切片，len=cap=5
    s := []int{11,22,33,44,55}
    fmt.Println(len(s), cap(s))
    fmt.Println(s)      //[11 22 33 44 55]
    fmt.Println(s[:3])  //[11 22 33]
    fmt.Println(s[2:])  //[33 44 55]
    fmt.Println(s[2:len(s)])    //[33 44 55]
    fmt.Println(s[1:2]) //[22]

    //前闭后开，冒号前的不包含，后面的包含。索引号，从0开始。

数组

长度属于数组类型的一部分，长度不同的数组属于不同的类型
数组是内置(build-in)类型,是一组同类型数据的集合，它是值类型，通过从0开始的下标索引访问元素值。在初始化后长度是固定的，无法修改其长度。当作为方法的参数传入时将复制一份数组而不是引用同一指针。数组的长度也是其类型的一部分，通过内置函数len(array)获取其长度。

package main

import "fmt"
func main() {
    //arr := [3]int{11,22,33}
    //arr := [3]int{}
    //arr := []int{0,0,0}
    arr := [...]int{0, 0, 0}
    arr[0] = 100
    arr[1] = 101
    arr[2] = 102
    fmt.Println(arr)
    fmt.Println(len(arr))

    for i := range arr {
        fmt.Printf("k=%v, v=%v\n", i, arr[i])
    }
    for k, v := range arr {
        fmt.Printf("k=%v, v=%v\n", k, v)
    }

    //只遍历获取key
    for i, _ := range arr {
        fmt.Printf("k=%v\n", i)
    }

    //只遍历获取value
    for _, i := range arr {
        fmt.Printf("v=%v\n", i)
    }

    arr2 := [4]int{1: 2, 3: 4}
    arr3 := [...]int{2: 22, 5: 55}
    var arr4 [3]int = [3]int{7, 8, 9}
    fmt.Println(arr2) //[0 2 0 4]
    fmt.Println(arr3) //[0 0 22 0 0 55]
    fmt.Println(arr4) //[7 8 9]

    //多维数组
    arr5 := [3][4]int{
        {1, 2, 3, 4},
        {11, 22, 33, 44},
        {1, 2, 3, 4}}
    fmt.Println(arr5) //[[1 2 3 4] [11 22 33 44] [1 2 3 4]]

    //创建长度为10的数组，最后一个元素是100
    fmt.Println([...]int{9: 100})

    //指向数组的指针
    p := new([3]int)
    var p1 = &[3]int{11, 22, 33}
    p4 := [5]int{}
    p5 := &p4
    fmt.Println(p)  //&[0 0 0]
    fmt.Println(p1) //&[11 22 33]
    fmt.Println(p5) //&[0 0 0 0 0]

    //指针数组
    var p2, p3 int = 101, 102
    a23 := [2]*int{&p2, &p3}
    fmt.Println(a23) //[0xc4200161a8 0xc4200161b0]

}
